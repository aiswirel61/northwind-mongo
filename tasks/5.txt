Return all products where product name starts with 'A', 'B', .... 'F' ordered by name.

| ProductId | ProductName | QuantityPerUnit |

db.products.aggregate([
    { $match : { ProductName: { $regex: '^[A-F]' } } },
    { $project : { 
        _id:false,
        ProductID: 1,
        ProductName: 1,
        QuantityPerUnit: 1
    }},
    { $sort:{ ProductName: 1 }}
])
