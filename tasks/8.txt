Create a query to return:

| CategoryName | TotalNumberOfProducts |

db.categories.aggregate([
    { $lookup: {
        from: 'products',
        localField: 'CategoryID',
        foreignField: 'CategoryID',
        as: 'Products'
    }},
    { $project: {
        _id: false,
        CategoryName: 1,
        TotalNumberOfProducts: {$size: "$Products"}
    }}
]);
